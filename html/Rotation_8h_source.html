82 {
83  const Vector3f& z = Vector3f::UnitZ();
84  const Vector3f zR = rotation.inverse() * z;
85  const Vector3f c = zR.cross(z);
86  const float sin = c.norm();
87  const float cos = zR.dot(z);

123 {
124  const Matrix3f mat = rot.normalized().toRotationMatrix();
125  const float m20 = mat(2, 0);
126 
127  if(std::abs(m20) < 0.999999f)
128  {
129  const float m00 = mat(0, 0);
130  const float m10 = mat(1, 0);
131  const float m21 = mat(2, 1);
132  const float m22 = mat(2, 2);
133 
134  const float y1 = -std::asin(m20);
135  const float y2 = pi - y1;
136  const float cy1 = std::cos(y1);
137  const float cy2 = std::cos(y2);
138  const float x1 = std::atan2(m21 / cy1, m22 / cy1);
139  const float x2 = std::atan2(m21 / cy2, m22 / cy2);
140  const float z1 = std::atan2(m10 / cy1, m00 / cy1);
141  const float z2 = std::atan2(m10 / cy2, m00 / cy2);
142 
143  Vector3f v1(x1, y1, z1);
144  Vector3f v2(x2, y2, z2);
145

172 inline float Rotation::Aldebaran::getXAngle(const Quaternionf& rot)
173 {
174  // | a b c | | a d g | | 0 | | g |
175  // A = | d e f | At = | b e h | At * | 0 | = | h | = v
176  // | g h i | | c f i | | 1 | | i |
177 
178  // Project | 0 |
179  // v into : v -> | h | = v'
180  // yz plane | i |
181 
182  // calculate angle | 0 | angleX = acos((v' * z) / (|v'| * |z|))
183  // bewteen v' and : z = | 0 |, <=> angleX = acos(i / |v'|)
184  // z-axis of A | 1 | <=> angleX = acos(i / sqrt(h * h + i * i))
185 
186  const float tx = 2.f * rot.x();
187  const float h = 2.f * rot.z() * rot.y() + tx * rot.w();
188  const float i = 1.f - (tx * rot.x() + 2.f * rot.y() * rot.y());
189 
190  const float len = std::sqrt(h * h + i * i);
191  return len > 1e-5f ? std::acos(i / len) * (h > 0.f ? 1.f : -1.f) : 0.f;
192 }
