36  float getDistanceToPoint(const Eigen::Vector2f& point, Eigen::Vector2f& closest_point) const {
37  // Return minimum distance between line segment vw and point p
38  const float l2 = (begin_ - end_).squaredNorm(); // i.e. |w-v|^2 - avoid a sqrt
39  if (l2 == 0.0f) return (begin_ - point).norm(); // v == w case
40  // Consider the line extending the segment, parameterized as v + t (w - v).
41  // We find projection of point p onto the line.
42  // It falls where t = [(p-v) . (w-v)] / |w-v|^2
43  // We clamp t from [0,1] to handle points outside the segment vw.
44  const float t = std::fmaxf(0.f, std::fminf(1.f, (point - end_).dot(begin_ - end_) / l2));
45  const Eigen::Vector2f projection = end_ + t * (begin_ - end_); // Projection falls on the segment
46  closest_point = projection;
47  return (point - projection).norm();
48  }
